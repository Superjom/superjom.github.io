Circle

判断是否有环，采用快慢双指针

def has_circle(head):
    if not head or not head.next: return False
    slow = head
    fast = head.next # Note here
    while fast and fast.next:
        if fast == slow: return True
      	fast = fast.next.next
        slow = slow.next
    return False
