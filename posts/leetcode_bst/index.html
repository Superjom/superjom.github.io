BST Generation

https://leetcode.com/problems/unique-binary-search-trees-ii/

def generate(n):
    def helper(l, r):
        trees = []
        if l == r:
            trees.append(TreeNode(l))
        elif l < r:
            for m in range(l, r+1): # note, there is n+1

                ls = helper(l, m-1)
                rs = helper(m+1, r)
                ls = ls if ls else [None]
                rs = rs if rs else [None]
                for lt in ls:
                    for rt in rs:
                        mid = TreeNode(m)
                        mid.left = lt
                        mid.right = rt
                        trees.append(mid)
         return trees
    return helper(1, n)
