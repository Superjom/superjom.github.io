链表常规操作（模板）

获取长度

def get_length(node):
    len = 0
    while node:
        len += 1
        node = node.next
    return len

拆分两半

采用 slow, fast 双指针，slow 指针每次走 1 步，fast 指针每次走 2 步

def split_two_half(node):
    pre = mid = fast = node

    while fast and fast.next:
        # fast pointer go 2x steps
        fast = fast.next.next
        pre = mid
        # mid pointer go x steps
        mid = mid.next

    l0 = node
    pre.next = None
    l2 = mid

    return l0, l2
